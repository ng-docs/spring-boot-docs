Annotation Type ConditionalOnEnabledResourceChain


  • @Target(value={TYPE,METHOD})
 @Retention(value=RUNTIME)
 @Documented
 @Conditional(value=org.springframework.boot.autoconfigure.web.OnEnabledResourceChainCondition.class)
public @interface ConditionalOnEnabledResourceChain
    @Conditional that checks whether or not the Spring resource handling chain is enabled. Matches if WebProperties.Resources.Chain.getEnabled() is true or if webjars-locator-core is on the classpath.
